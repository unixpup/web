And here's some TypeScript:

```typescript
interface Post {
  id: string;
  title: string;
  content: string;
  publishDate: Date;
  tags: string[];
}

type PostStatus = 'draft' | 'published' | 'archived';

class BlogManager {
  private posts: Map<string, Post & { status: PostStatus }> = new Map();

  addPost(post: Post, status: PostStatus = 'draft'): void {
    this.posts.set(post.id, { ...post, status });
  }

  publishPost(id: string): boolean {
    const post = this.posts.get(id);
    if (!post) return false;

    this.posts.set(id, { ...post, status: 'published' });
    return true;
  }

  getPublishedPosts(): Post[] {
    return Array.from(this.posts.values())
      .filter(post => post.status === 'published');
  }
}
```

## Working with MDX

MDX allows us to do some pretty cool things beyond just standard markdown. For example, we can:

- Import and use React components
- Export variables
- Use JSX expressions

Here's a simple example of a React component:

```jsx
import React, { useState } from 'react';

export const Counter = () => {
  const [count, setCount] = useState(0);

  return (
    <div className="counter">
      <p>Count: {count}</p>
      <button onClick={() => setCount(count + 1)}>Increment</button>
      <button onClick={() => setCount(count - 1)}>Decrement</button>
    </div>
  );
};
```
